Golang Postgresql

  1. 测试用例：
CREATE TABLE users (
id uuid primary key default gen_random_uuid(),
mobile text UNIQUE not null
);
INSERT INTO users (mobile) VALUES ('86-1000000000')
  1. 全在代码里：
package main

import (
    "database/sql"
    "log"

    //"fmt"
    "fmt"
    _ "github.com/lib/pq"
    "time"
    "github.com/lib/pq"
)

func main() {

    // 数据库连接
    db, err := sql.Open("postgres", "user=postgres dbname=meidb sslmode=disable")
    if err != nil {
        log.Fatal(err)
    }

    // 连接数
    db.SetMaxOpenConns(10) // 最大连接数
    db.SetMaxIdleConns(5)  // 最大空闲数

    for i := 0; i < 10; i++ {
        go func() {
            db.Ping()
        }()
    }
    //time.Sleep(10 * time.Second)

    // Exec 执行一次，没有具体值返回
    a, err := HowToExec(db)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("Exec Result:", a)

    // Prepare
    for i := 0; i < 10; i++ {
        go func() {
            p, err := HowToPrepare(db)
            if err != nil {
                log.Fatal(err)
            }
            fmt.Println("Prepare Result:", p)
        }()
    }
    time.Sleep(20 * time.Second)

    // QueryRow
    result, err := HowToQueryRow(db)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println("QueryRow Result:", result)

    // Query
    total, results := HowToQuery(db)
    fmt.Println("Query Result:", total, results)


    // Begin
    HowToBegin(db)

    // 关闭连接
    defer db.Close()

}

func HowToExec(db *sql.DB) (int64, error) {
    m := "86-1000000000"
    q := "UPDATE users SET mobile=$1 WHERE mobile=$1;"
    ret, err := db.Exec(q, m)
    if err != nil {
        return 0, err
    }
    r, err := ret.RowsAffected()
    ret.LastInsertId()
    if err != nil {
        return 0, err
    }
    return r, nil
}

func HowToPrepare(db *sql.DB) ([]string, error) {
    m := "86-1000000000"
    q := "SELECT id FROM users WHERE mobile=$1"
    sm, err := db.Prepare(q)
    defer sm.Close()
    if err != nil {
        return []string{}, err
    }
    exec, err := sm.Exec(m)
    if err != nil {
        log.Println(err)
        return []string{}, nil
    }
    ret, err := exec.RowsAffected()
    if err != nil {
        log.Println(err)
        return []string{}, nil
    }
    fmt.Println("Prepare Exec Result:", ret)

    var queryTemp string
    query := sm.QueryRow(m)
    err = query.Scan(&queryTemp)
    if err != nil {
        log.Println(err)
        return []string{}, nil
    }
    fmt.Println("Prepare QueryRow Result:", queryTemp)

    sq, err := sm.Query(m)
    defer sq.Close()
    var re []string
    for sq.Next() {
        var tmp string
        err := sq.Scan(&tmp)
        if err != nil {
            log.Println(err)
            return []string{}, nil
        }
        re = append(re, tmp)
    }
    fmt.Println("Prepare Query Result:", re)

    return re, nil
}

func HowToQueryRow(db *sql.DB) (string, error) {
    m := "86-1000000000"
    var temp string
    row := db.QueryRow("SELECT id FROM users WHERE mobile=$1", m)
    err := row.Scan(&temp)
    if err != nil {
        if err == sql.ErrNoRows {
            // 没有查询到结果的返回,可以进行日志输出
            return temp, nil
        }
        // 出错，但是不是查询结果的问题
        return temp, err
    }
    return temp, nil
}

func HowToQuery(db *sql.DB) (total int, result []string) {
    q := "SELECT id FROM users WHERE mobile=$1 ORDER BY id DESC OFFSET $2 LIMIT $3;"
    qa := "SELECT count(*) FROM users WHERE mobile=$1;"
    m := "86-1000000000"
    var temp string

    // 查询总数
    row := db.QueryRow(qa, m)
    err := row.Scan(&total)
    if err != nil {
        if err == sql.ErrNoRows {
            return 0, []string{}
        }
        log.Fatal(err)
    }

    // 结果查询
    rows, err := db.Query(q, m, 0, 20)
    defer rows.Close() // 将query关闭之后才能将连接放回连接池
    if err != nil {
        log.Fatal(err)
    }
    // 依次的准备好每一条查出的数据
    for rows.Next() {
        err := rows.Scan(&temp)
        if err != nil {
            log.Fatal(err)

        }
        result = append(result, temp)
    }
    return total, result
}
